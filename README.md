```python
import numpy as np  # –ë–∏–±–ª–∏–æ—Ç–µ–∫–∞ –¥–ª—è —Ä–∞–±–æ—Ç—ã —Å –º–∞—Å—Å–∏–≤–∞–º–∏ –∏ –º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–∏–º–∏ –≤—ã—á–∏—Å–ª–µ–Ω–∏—è–º–∏
import time  # –î–ª—è —Ä–∞–±–æ—Ç—ã —Å–æ –≤—Ä–µ–º–µ–Ω–µ–º –∏ –∑–∞–¥–µ—Ä–∂–∫–∞–º–∏
import sys  # –î–ª—è –≤–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏—è —Å —Å–∏—Å—Ç–µ–º–æ–π (–≤—ã–≤–æ–¥ –≤ —Ç–µ—Ä–º–∏–Ω–∞–ª)
import math  # –î–ª—è –º–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–∏—Ö —Ñ—É–Ω–∫—Ü–∏–π (sin, cos)
from collections import deque  # –î–≤—É—Å—Ç–æ—Ä–æ–Ω–Ω—è—è –æ—á–µ—Ä–µ–¥—å –¥–ª—è –ø–æ–¥—Å—á–µ—Ç–∞ FPS
import colorsys  # –î–ª—è —Ä–∞–±–æ—Ç—ã —Å —Ü–≤–µ—Ç–æ–≤—ã–º–∏ –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–∞–º–∏ (HSV –≤ RGB)


def rotate_points(points, angle_y):
    # –°–æ–∑–¥–∞–µ–º –º–∞—Ç—Ä–∏—Ü—É –ø–æ–≤–æ—Ä–æ—Ç–∞ –≤–æ–∫—Ä—É–≥ –æ—Å–∏ Y
    Ry = np.array([
        [np.cos(angle_y), 0, np.sin(angle_y)],     # –ü–µ—Ä–≤–∞—è —Å—Ç—Ä–æ–∫–∞ –º–∞—Ç—Ä–∏—Ü—ã
        [0, 1, 0],                                 # –í—Ç–æ—Ä–∞—è —Å—Ç—Ä–æ–∫–∞ –º–∞—Ç—Ä–∏—Ü—ã
        [-np.sin(angle_y), 0, np.cos(angle_y)]     # –¢—Ä–µ—Ç—å—è —Å—Ç—Ä–æ–∫–∞ –º–∞—Ç—Ä–∏—Ü—ã
    ])
    # –£–º–Ω–æ–∂–∞–µ–º —Ç–æ—á–∫–∏ –Ω–∞ —Ç—Ä–∞–Ω—Å–ø–æ–Ω–∏—Ä–æ–≤–∞–Ω–Ω—É—é –º–∞—Ç—Ä–∏—Ü—É –ø–æ–≤–æ—Ä–æ—Ç–∞
    return np.dot(points, Ry.T)


def calculate_fps(fps_counter):
    if len(fps_counter) < 2:  # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞–ª–∏—á–∏—è –º–∏–Ω–∏–º—É–º –¥–≤—É—Ö –≤—Ä–µ–º–µ–Ω–Ω—ã—Ö –º–µ—Ç–æ–∫
        return 0.0
    
    
    # –í—ã—á–∏—Å–ª—è–µ–º —Ä–∞–∑–Ω–∏—Ü—É –≤—Ä–µ–º–µ–Ω–∏ –º–µ–∂–¥—É –ø–µ—Ä–≤–æ–π –∏ –ø–æ—Å–ª–µ–¥–Ω–µ–π –º–µ—Ç–∫–æ–π
    time_diff = fps_counter[-1] - fps_counter[0]
    

    if time_diff <= 0:  # –ó–∞—â–∏—Ç–∞ –æ—Ç –¥–µ–ª–µ–Ω–∏—è –Ω–∞ –Ω–æ–ª—å
        return 0.0


    # –í–æ–∑–≤—Ä–∞—â–∞–µ–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∫–∞–¥—Ä–æ–≤ / –≤—Ä–µ–º—è = FPS
    return len(fps_counter) / time_diff


def get_colored_char(char, hue, saturation=1.0, value=1.0):
    # –û–≥—Ä–∞–Ω–∏—á–∏–≤–∞–µ–º –æ—Ç—Ç–µ–Ω–æ–∫ –≤ –¥–∏–∞–ø–∞–∑–æ–Ω–µ [0, 1]
    hue = max(0.0, min(1.0, hue))
    

    # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º HSV –≤ RGB, –º–∞—Å—à—Ç–∞–±–∏—Ä—É–µ–º –¥–æ 255
    r, g, b = [int(x * 255) for x in colorsys.hsv_to_rgb(hue, saturation, value)]
    

    # –§–æ—Ä–º–∏—Ä—É–µ–º ANSI-–ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å –¥–ª—è —Ü–≤–µ—Ç–Ω–æ–≥–æ —Å–∏–º–≤–æ–ª–∞
    return f"\033[38;2;{r};{g};{b}m{char}\033[0m"


def create_heart_points(scale=5, num_points=1000, num_layers=30):
    # –°–æ–∑–¥–∞–µ–º –ø–∞—Ä–∞–º–µ—Ç—Ä–∏—á–µ—Å–∫—É—é –∫—Ä–∏–≤—É—é
    t = np.linspace(0, 2*np.pi, num_points)
    

    # –ü–∞—Ä–∞–º–µ—Ç—Ä–∏—á–µ—Å–∫–∏–µ —É—Ä–∞–≤–Ω–µ–Ω–∏—è —Å–µ—Ä–¥—Ü–∞
    x = 16 * np.sin(t)**3
    y = 13 * np.cos(t) - 5 * np.cos(2*t) - 2 * np.cos(3*t) - np.cos(4*t)
    z = np.zeros_like(x)

    points = []


    # –°–æ–∑–¥–∞–µ–º —Å–ª–æ–∏ –¥–ª—è –æ–±—ä–µ–º–∞
    for i in range(num_layers):
        factor = 1 - i/num_layers
        layer_x = factor * x
        layer_y = factor * y
        layer_z = np.full_like(x, -i/2)
        points.extend(zip(layer_x, layer_y, layer_z))


    # –î–æ–±–∞–≤–ª—è–µ–º –≤–Ω—É—Ç—Ä–µ–Ω–Ω–∏–µ —Ç–æ—á–∫–∏
    for _ in range(num_points // 2):
        r = np.random.random() * 0.8
        theta = np.random.random() * 2 * np.pi
        phi = np.random.random() * np.pi
        

        # –í—ã—á–∏—Å–ª—è–µ–º –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã –≤–Ω—É—Ç—Ä–µ–Ω–Ω–∏—Ö —Ç–æ—á–µ–∫
        x = r * 16 * np.sin(theta)**3 * np.sin(phi)
        y = r * (13 * np.cos(theta) - 5 * np.cos(2*theta) - 
                2 * np.cos(3*theta) - np.cos(4*theta)) * np.sin(phi)
        z = r * 15 * np.cos(phi)
        points.append((x, y, z))


    # –ú–∞—Å—à—Ç–∞–±–∏—Ä—É–µ–º –≤—Å–µ —Ç–æ—á–∫–∏
    return scale * np.array(points)



def draw_heart(points, width=80, height=40, time_val=0):
    # –°–∏–º–≤–æ–ª—ã –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è –≥–ª—É–±–∏–Ω—ã
    shading_chars = " .:!*OQ#"
    
    # –ü—Ä–æ–µ—Ü–∏—Ä—É–µ–º 3D –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã –Ω–∞ 2D —ç–∫—Ä–∞–Ω
    x = (points[:, 0] / np.max(np.abs(points[:, 0])) * (width//2) + width//2).astype(int)


    # –ò–∑–º–µ–Ω—è–µ–º –∑–Ω–∞–∫ –¥–ª—è y-–∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã, —á—Ç–æ–±—ã –ø–µ—Ä–µ–≤–µ—Ä–Ω—É—Ç—å —Å–µ—Ä–¥—Ü–µ
    y = (-points[:, 1] / np.max(np.abs(points[:, 1])) * (height//2) + height//2).astype(int)

    z = points[:, 2]
    

    # –û—Ç—Å–µ–∫–∞–µ–º —Ç–æ—á–∫–∏ –∑–∞ –ø—Ä–µ–¥–µ–ª–∞–º–∏ —ç–∫—Ä–∞–Ω–∞
    mask = (0 <= x) & (x < width) & (0 <= y) & (y < height)
    x, y, z = x[mask], y[mask], z[mask]
    

    # –°–æ–∑–¥–∞–µ–º —ç–∫—Ä–∞–Ω –∏ z-–±—É—Ñ–µ—Ä
    screen = np.full((height, width), ' ', dtype=object)
    z_buffer = np.full((height, width), float('-inf'))
    

    # –û—Ç—Ä–∏—Å–æ–≤–∫–∞ —Ç–æ—á–µ–∫ —Å —É—á–µ—Ç–æ–º –≥–ª—É–±–∏–Ω—ã
    if len(z) > 0:
        z_min, z_max = np.min(z), np.max(z)
        if z_max > z_min:
            z_normalized = (z - z_min) / (z_max - z_min)
            intensity = (z_normalized * (len(shading_chars) - 1)).astype(int)
            
            for xi, yi, zi, char_index in zip(x, y, z, intensity):
                if zi > z_buffer[yi, xi]:
                    z_buffer[yi, xi] = zi
                    z_factor = (zi - z_min) / (z_max - z_min)
                    hue = (time_val + z_factor) % 1.0
                    screen[yi, xi] = get_colored_char(shading_chars[char_index], hue)
    
    return '\n'.join(''.join(row) for row in screen)



def pulsating_effect(time):
    # –°–æ–∑–¥–∞–µ—Ç —Å–∏–Ω—É—Å–æ–∏–¥–∞–ª—å–Ω–æ–µ –∏–∑–º–µ–Ω–µ–Ω–∏–µ —Ä–∞–∑–º–µ—Ä–∞
    return 1 + 0.05 * math.sin(time * 2)



def main():
    # –°–æ–∑–¥–∞–µ–º –Ω–∞—á–∞–ª—å–Ω—ã–µ —Ç–æ—á–∫–∏ —Å–µ—Ä–¥—Ü–∞
    heart_points = create_heart_points(scale=8)
    angle_y = 0
    

    # –û—á–∏—â–∞–µ–º —ç–∫—Ä–∞–Ω –∏ —Å–∫—Ä—ã–≤–∞–µ–º –∫—É—Ä—Å–æ—Ä
    print('\033[2J')
    print('\033[?25l')
    

    # –°–æ–∑–¥–∞–µ–º –æ—á–µ—Ä–µ–¥—å –¥–ª—è –ø–æ–¥—Å—á–µ—Ç–∞ FPS
    fps_counter = deque(maxlen=60)
    start_time = time.time()
    
    
    try:
        while True:
            # –í—Ä–µ–º—è –Ω–∞—á–∞–ª–∞ –∫–∞–¥—Ä–∞
            frame_start = time.time()
            current_time = frame_start - start_time
            

            # –ú–∞—Å—à—Ç–∞–±–∏—Ä—É–µ–º —Å–µ—Ä–¥—Ü–µ
            scale = pulsating_effect(current_time)
            scaled_points = heart_points * scale
            
            
            # –ü–æ–≤–æ—Ä–∞—á–∏–≤–∞–µ–º —Å–µ—Ä–¥—Ü–µ
            rotated_points = rotate_points(scaled_points, angle_y)
            

            # –û—Ç—Ä–∏—Å–æ–≤—ã–≤–∞–µ–º —Å–µ—Ä–¥—Ü–µ
            frame = draw_heart(rotated_points, time_val=(current_time * 0.1) % 1.0)
            

            # –ü–æ–¥—Å—á–∏—Ç—ã–≤–∞–µ–º FPS
            fps_counter.append(time.time())
            fps = calculate_fps(fps_counter)
            

            # –í—ã–≤–æ–¥–∏–º FPS –∏ —Å–µ—Ä–¥—Ü–µ
            status_line = f"\033[1mFPS: {fps:.1f} | Press Ctrl+C to exit\033[0m"
            frame_with_status = frame + "\n" + status_line
            

            # –¶–µ–Ω—Ç—Ä–∏—Ä—É–µ–º –≤—ã–≤–æ–¥
            terminal_width = 80
            frame_width = len(frame.split('\n')[0])
            padding = ' ' * ((terminal_width - frame_width) // 2)
            frame_with_status = padding + frame_with_status + padding
            

            # –í—ã–≤–æ–¥–∏–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç
            sys.stdout.write('\033[H' + frame_with_status)
            sys.stdout.flush()
            

            # –û–±–Ω–æ–≤–ª—è–µ–º —É–≥–æ–ª –ø–æ–≤–æ—Ä–æ—Ç–∞
            angle_y += 0.05
            

            # –û–≥—Ä–∞–Ω–∏—á–∏–≤–∞–µ–º FPS
            frame_time = time.time() - frame_start
            if frame_time < 0.033:
                time.sleep(0.033 - frame_time)


    except KeyboardInterrupt:
        # –í–æ—Å—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –∫—É—Ä—Å–æ—Ä –∏ –æ—á–∏—â–∞–µ–º —ç–∫—Ä–∞–Ω
        print('\033[?25h')
        print("\nProgram terminated")
        

if __name__ == "__main__":
    try:
        main()
    except KeyboardInterrupt:
        print('\033[?25h')
        print("\nProgram terminated")
